create table Users
(
    userId int primary key identity(1,1),
    firstName varchar(255),
    lastName varchar(255),
    email varchar(255),
    password varchar(255),
    phone int,
    speciality varchar(255),
    gender varchar(255),
    bloodGroup varchar(255),
    age int,
    city varchar(255),
    userType int,
    address  varchar(255),
    storeName varchar(255),
	latitude int,
	longitude int,
    deleted bit
)
